Q1. 다음과 같이 테이블을 작성하시오.
사용DB : mbasic
테이블만들기 : userinfo_repeat
    번호 : no      자료형 int   필수입력 , 숫자자동증가, 기본키
    이름 : name  자료형 varchar(10)  필수입력
    나이 : age    자료형  int    필수입력

 use mbasic
 create table userinfo_repeat(
   no                  int not null  auto_increment primary key, 
   name  varchar(10) not null,
   age                 int not null
 );
